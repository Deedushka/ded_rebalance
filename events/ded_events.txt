country_event = {
	id = debug_ded.1
	is_triggered_only = yes
	hidden = yes

	immediate = {
		event_target:lobotomy_patient = {
			if = {
				limit = {
					NOT = { is_in_array = { array = global.playable_countries value = THIS } }
					NOT = { is_in_array = { array = global.important_countries value = THIS } }
					if = { limit = { has_global_flag = AAL_SOV_collapse } AAL_collapse_countries = no }
					is_ai = yes
				}
				set_country_flag = lobotomized
				add_ideas = ded_lobotomized

				destroy_ships = {
					type = ship_hull_light
					count = all
				}
				destroy_ships = {
					type = ship_hull_cruiser
					count = all
				}
				destroy_ships = {
					type = ship_hull_heavy
					count = all
				}
				destroy_ships = {
					type = ship_hull_carrier 
					count = all
				}
				destroy_ships = {
					type = ship_hull_submarine
					count = all
				}

				every_controlled_state = {
					limit = { is_controlled_by = THIS }
					PREV = {
						delete_unit = {
							state = PREV
						}
					}
				}
			}
		}
	}
}

country_event = {	# Peace conference (puppet)
	id = debug_ded_pc.1
	is_triggered_only = yes
	hidden = yes

	immediate = { 
		event_target:pc_winner = {
			white_peace = event_target:pc_loser
			if = {
				limit = {
					NOT = {
						has_government = democratic
					}
				}
				set_autonomy = {
					target = event_target:pc_loser
					autonomy_state = autonomy_integrated_puppet
				}
			}
			else = {
				set_autonomy = {
					target = event_target:pc_loser
					autonomy_state = autonomy_supervised_state
				}
			}
		}
		event_target:pc_loser = {
			country_event = { id = debug_ded_pc.10 days = 1 }
		}
	}
}
