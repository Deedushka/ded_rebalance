	# Check guarantees on Lit
	on_monthly_POL = {
		effect = {
			if = {
				limit = {
					has_guaranteed = LIT
					has_capitulated = no
				}
				LIT = {
					diplomatic_relation = {
						country = POL
						relation = guarantee
						active = no
					}
				}
				POL = {
					diplomatic_relation = {
						country = LIT
						relation = guarantee
						active = no
					}
				}
			}
		}
	}

	# Lobotomize Ai
	on_monthly = {
		effect = {
			if = {
				limit = {
					has_game_rule = {
						rule = OFF_NON_IMPORTANT_AI
						option = OFF
					}
				}
				if = {
					limit = {
						is_ai = yes
						OR = {
							NOT = { is_in_array = { array = global.playable_countries value = THIS } }
							NOT = { is_in_array = { array = global.important_countries value = THIS } }
							has_country_flag = lobotomized
						}
					}
					save_event_target_as = lobotomy_patient
					country_event = debug_ded.1
				}
				else_if = {
					limit = {
						has_country_flag = lobotomized
						is_ai = no
					}
					save_event_target_as = lobotomy_patient
					country_event = debug_ded.2
				}
			}
		}
	}

	# No peace conference
	on_before_peace_conference_start = {
