on_actions = {
	on_startup = {
		effect = {
			# Playable countries array
			every_country = {
				limit = { 
					OR = {
						original_tag = GER
						original_tag = ITA

						original_tag = JAP
						original_tag = SIA
	
						original_tag = HUN
						original_tag = ROM
						original_tag = BUL
	
						original_tag = ENG
						original_tag = FRA
						original_tag = USA
					
						original_tag = CAN
						original_tag = SAF
						original_tag = RAJ
						original_tag = AST
						original_tag = NZL
					
						original_tag = SOV
					}
				}
				add_to_array = {
					array = global.playable_countries
					value = THIS
				}
			}
			# Important AI
			every_possible_country = {
				limit = {
					OR = {
						original_tag = SPR
						original_tag = BEL
						original_tag = HOL
						original_tag = FIN
						original_tag = POL
						original_tag = YUG
						original_tag = GRE
						original_tag = NOR
						original_tag = DEN

						original_tag = ETH
						original_tag = IRQ
						original_tag = PER

						original_tag = CHI
						original_tag = MAN
						is_chinese_warlord = yes
						original_tag = PRC

						original_tag = D01
						original_tag = D02
						original_tag = D03
						original_tag = D04
						original_tag = D05
					}
				}
				add_to_array = {
					array = global.important_countries
					value = THIS
				}
			}
		}

		# Extension
		effect = {
			GER = {
				if = {
					limit = {
						has_game_rule = {
							rule = INTEGRATION_COUNTRY_TO_OVERLORD_AXIS
							option = INTEGRATION
						}	
					}
					for_each_scope_loop = {
						array = global.playable_countries
						if = {
							limit = {
								OR = {
									original_tag = HUN
									original_tag = BUL
									original_tag = ROM
								}
							}
							if = { 
								limit = { is_ai = yes } 
								THIS = {
									add_ideas = economy_integration_axis
								}
							}
						}
					}
				}
			}
			ENG = {
				if = {
					limit = {
						has_game_rule = {
							rule = INTEGRATION_COUNTRY_TO_OVERLORD_ALLIES
							option = INTEGRATION
						}
					}
					for_each_scope_loop = {
						array = global.playable_countries
						if = {
							limit = {
								OR = {
									original_tag = CAN
									original_tag = SAF
									original_tag = RAJ
									original_tag = AST
									original_tag = NZL
								}
							}
							if = { 
								limit = { is_ai = yes }
								THIS = {
									add_ideas = economy_integration_allies
								}
							}
						}
					}
				}
				add_timed_idea = {
					idea = extra_planes_cost
					years = 3
				}
			}

			# No embargo
			every_country = {
				limit = {
					OR = {
						original_tag = SWE
						original_tag = TUR
						original_tag = POR
						original_tag = PER
					}
				}
				add_ideas = no_embargo_modifier
			}
			for_each_scope_loop = {
				array = global.countries
				if = { 
					limit = { is_in_array = { array = global.playable_countries value = THIS } } 
					army_experience = 30
					navy_experience = 30
					air_experience = 20
				}
			}
		}

		# Lobotomize Ai
		effect = {
			if = {
				limit = {
					has_game_rule = {
						rule = OFF_NON_IMPORTANT_AI
						option = OFF
					}
				}
				every_country = {
					save_event_target_as = lobotomy_patient
					country_event = debug_ded.1
				}
			}
		}

		# Annex China
		effect = {
			if = {
				limit = {
					NOT = {
						has_game_rule = {
							rule = ANNEX_ALL_CHINA
							option = NOT_ANNEX
						}
					}
				}
				every_country = {
					limit = { OR = { is_chinese_warlord = yes original_tag = PRC } }
					CHI = {  
						annex_country = {
							target = PREV
						}
					}
				}
				CHI = {
					add_ideas = china_easy_cap
				}
			}
		}
	}

	# China cap factory count
	on_capitulation = {
		effect = {
			if = {
				limit = {
					original_tag = CHI
					NOT = {
						has_game_rule = {
							rule = ANNEX_ALL_CHINA
							option = NOT_ANNEX
						}
					}
				}
				618 = {
					add_resource = {
						type = oil
						amount = 45
					}
				}
				744 = {
					add_extra_state_shared_building_slots = 10
					add_building_construction = {
						type = industrial_complex
						level = 10
						instant_build = yes
					}
				}
				283 = {
					add_extra_state_shared_building_slots = 10
					add_building_construction = {
						type = industrial_complex
						level = 10
						instant_build = yes
					}
				}
				1032 = {
					add_extra_state_shared_building_slots = 10
					add_building_construction = {
						type = industrial_complex
						level = 10
						instant_build = yes
					}
				}
				602 = {
					add_extra_state_shared_building_slots = 9
					add_building_construction = {
						type = arms_factory
						level = 9
						instant_build = yes
					}
				}
				600 = {
					add_extra_state_shared_building_slots = 9
					add_building_construction = {
						type = arms_factory
						level = 9
						instant_build = yes
					}
				}
				749 = {
					add_extra_state_shared_building_slots = 9
					add_building_construction = {
						type = arms_factory
						level = 9
						instant_build = yes
					}
				}
			}
		}
	}

	# Japan-China war
	on_declare_war = {
		effect = {
			if = {
				limit = { 
					tag = JAP
					FROM = { tag = CHI }
				}
				JAP = { set_rule = { can_puppet = no } }
			}
		}
	}
	on_peace = {
		effect = {
			if = {
				limit = { 
					original_tag = CHI
				}
				JAP = { set_rule = { can_puppet = yes } }
			}
		}
	}

	# Check guarantees on Lit
	on_monthly_POL = {
		effect = {
			if = {
				limit = {
					has_guaranteed = LIT
					has_capitulated = no
				}
				LIT = {
					diplomatic_relation = {
						country = POL
						relation = guarantee
						active = no
					}
				}
				POL = {
					diplomatic_relation = {
						country = LIT
						relation = guarantee
						active = no
					}
				}
			}
		}
	}

	# Lobotomize Ai
	on_monthly = {
		effect = {
			if = {
				limit = {
					has_game_rule = {
						rule = OFF_NON_IMPORTANT_AI
						option = OFF
					}
				}
				if = {
					limit = {
						is_ai = yes
						OR = {
							NOT = { is_in_array = { array = global.playable_countries value = THIS } }
							NOT = { is_in_array = { array = global.important_countries value = THIS } }
							has_country_flag = lobotomized
						}
					}
					save_event_target_as = lobotomy_patient
					country_event = debug_ded.1
				}
				else_if = {
					limit = {
						has_country_flag = lobotomized
						is_ai = no
					}
					save_event_target_as = lobotomy_patient
					country_event = debug_ded.2
				}
			}
		}
	}
}
