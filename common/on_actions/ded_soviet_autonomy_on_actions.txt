on_actions = {
	on_capitulation = {
		effect = {
			if = {
				limit = {
					original_tag = UKR
				}
				SOV = { country_event = soviet_autonomy_cap.1 }
			}
		}
	}
	on_startup = {
		effect = {
			every_country = {
				limit = {
					is_soviet_autonomy = yes
				}
				every_controlled_state = {
					limit = {
						state_has_any_resource = yes
					}
					PREV = {
						give_resource_rights = {
							receiver = OVERLORD
							state = PREV
						}
						# log = "GET NAMES NOW: [?THIS.GetName] [?ROOT.GetName] [?FROM.GetName] [?PREV.GetName]"
					}
				}
			}
		}
		effect = {
			every_state = {
				limit = {
					is_UKR_Insurgent_army = yes
				}
				every_country = {
					limit = {
						PREV = {
							is_core_of = PREV
						}
					}
					PREV = {
						remove_core_of = PREV
					}
					add_to_array = {
						array = PREV.core_array
						value = THIS
					}
				}
				get_dynamic_population_partisans_on_states = yes
				add_dynamic_modifier = {
					modifier = UKR_Insurgent_Army
				}
				add_core_of = UPA
				start_resistance = UPA
				add_resistance = 30
				add_compliance = 50
				add_to_array = {
					array = global.is_UKR_Insurgent_army
				}
			}
		}
	}
}
