on_actions = {
	on_war_relation_added = {
		effect = {
			if = {
				limit = {
					OR = {
						AND = {
							ROOT = { original_tag = GER }
							FROM = { original_tag = SOV }
						}
						AND = {
							ROOT = { original_tag = SOV }
							FROM = { original_tag = GER }
						}
					}
					GER = { 
						NOT = {
							has_country_flag = GER_win_barb
						}
					}
				}
				GER = { 
					activate_AAL_missions = yes
				}
				SOV = {
					activate_AAL_missions_spectate = yes
					if = {
						limit = {
							NOT = { has_completed_focus = SOV_claims_in_baltic }
						}
						complete_national_focus = SOV_claims_in_baltic
					}
				}
			}
		}
		effect = {
			if = {
				limit = {
					GER = { has_active_mission = AAL_barbarossa_mission }
					OR = {
						AND = {
							ROOT = { original_tag = JAP }
							FROM = { original_tag = SOV }
						}
						AND = {
							ROOT = { original_tag = SOV }
							FROM = { original_tag = JAP }
						}
					}
				}
				JAP = { set_country_flag = JAP_in_barb }
			}
		}
	}
	on_state_control_changed = {
		effect = {
			if = {
				limit = {
					FROM.FROM = { state = 219 }
					ROOT = {
						has_active_mission = AAL_barbarossa_mission
					}
				}
				FROM = {
					kill_country_leader = yes
					SOV_promote_to_country_leader_beriya = yes
				}
			}
		}
	}
	on_capitulation_immediate = {
		effect = {
			if = {
				limit = {
					ROOT = {
						AAL_collapse_countries = yes
					}
				}
				if = {
					limit = {
						ROOT = { original_tag = VLA }
					}
					annex_country = {
						target = TAY
						transfer_troops = no
					}
					annex_country = {
						target = EVE
						transfer_troops = no
					}
					annex_country = {
						target = YAK
						transfer_troops = no
					}
				}
				FROM = {
					annex_country = {
						target = ROOT
						transfer_troops = no
					}
				}
			}
		}
	}
	on_annex = {
		effect = {
			if = {
				limit = {
					FROM = {
						AAL_collapse_countries = yes
					}
				}
				FROM = {
					every_core_state = {
						add_core_of = ROOT
					}
				}
			}
		}
	}
	on_war = {
		effect = {
			if = {
				limit = {
					has_autonomy_state = autonomy_soviet
					has_defensive_war = yes
				}
				add_ideas = autonomy_soviet_partisans
			}
		}
	}
	on_peace = {
		effect = {
			if = {
				limit = { 
					has_autonomy_state = autonomy_soviet
					has_idea = autonomy_soviet_partisans
				}
				remove_ideas = autonomy_soviet_partisans
			}
		}
	}
}
