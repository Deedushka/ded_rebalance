on_actions = {
	on_war_relation_added = {
		effect = {
			if = {
				limit = {
					NOT = {
						has_country_flag = GER.GER_win_AAL
					}
					OR = {
						AND = {
							ROOT = { original_tag = GER }
							FROM = { original_tag = SOV }
						}
						AND = {
							ROOT = { original_tag = SOV }
							FROM = { original_tag = GER }
						}
					}
				}
				set_global_flag = AAL_been_continued
				every_state = {
					limit = {
						AAL_Objective_timed = yes
						CONTROLLER = {
							OR = {
								original_tag = SOV
								is_subject_of = SOV
							}
						}
					}
					add_to_array = {
						array = global.AAL_Objective_timed
					}
				}
				GER = {
					add_ideas = {
						AAL_GER_barbarossa
					}
				}
			}
		}
		effect = {
			if = {
				limit = {
					has_global_flag = AAL_been_continued
					OR = {
						AND = {
							ROOT = { original_tag = JAP }
							FROM = { original_tag = SOV }
						}
						AND = {
							ROOT = { original_tag = SOV }
							FROM = { original_tag = JAP }
						}
					}
				}
				set_country_flag = JAP.JAP_in_AAL
			}
		}
	}
	on_state_control_changed = {
		effect = {
			if = {
				limit = {
					FROM.FROM = { state = 219 }
					has_global_flag = AAL_been_continued
				}
				FROM = {
					kill_country_leader = yes
					SOV_promote_to_country_leader_beriya = yes
				}
			}
		}
		effect = {
			if = {
				limit = {
					is_in_array = {
						array = global.AAL_Objective_timed
						value = FROM.FROM.FROM
					}
				}
				if = {
					limit = {
						ROOT = {
							is_GER_Axis_controlled = yes
						}
					}
					GER = {
						add_days_mission_timeout = {
							mission = AAL_barbarossa_mission
							days = 21
						}
					}
				}
				else = {
					GER = {
						add_days_mission_timeout = {
							mission = AAL_barbarossa_mission
							days = -21
						}
					}
				}
			}
		}
	}
	on_capitulation_immediate = {
		effect = {
			if = {
				limit = {
					ROOT = {
						AAL_collapse_countries = yes
					}
				}
				if = {
					limit = {
						ROOT = { original_tag = VLA }
					}
					annex_country = {
						target = TAY
						transfer_troops = no
					}
					annex_country = {
						target = EVE
						transfer_troops = no
					}
					annex_country = {
						target = YAK
						transfer_troops = no
					}
				}
				FROM = {
					annex_country = {
						target = ROOT
						transfer_troops = no
					}
				}
			}
		}
	}
	on_annex = {
		effect = {
			if = {
				limit = {
					FROM = {
						AAL_collapse_countries = yes
					}
				}
				FROM = {
					every_core_state = {
						add_core_of = ROOT
					}
				}
			}
		}
	}
	on_war = {
		effect = {
			if = {
				limit = {
					has_autonomy_state = autonomy_soviet
					has_defensive_war = yes
				}
				add_ideas = autonomy_soviet_partisans
			}
		}
	}
	on_peace = {
		effect = {
			if = {
				limit = { 
					has_autonomy_state = autonomy_soviet
					has_idea = autonomy_soviet_partisans
				}
				remove_ideas = autonomy_soviet_partisans
			}
		}
	}
}
