@subtraction_var = 0.02
set_UKR_Insurgent_Army_down = {
	if = {
		limit = {
			NOT = {
				AND = {
					check_variable = { UKR_Insurgent_Army_local_org_regain = 0 }
					check_variable = { UKR_Insurgent_Army_compliance_growth = 0 }
					check_variable = { UKR_Insurgent_Army_resistance_growth = 0 }
					check_variable = { UKR_Insurgent_Army_resistance_target = 0 }
					check_variable = { UKR_Insurgent_Army_resistance_decay = 0 }
					check_variable = { UKR_Insurgent_Army_resistance_damage_to_garrison = 0 }
				}
			}
		}
		if = {
			limit = {
				check_variable = { UKR_Insurgent_Army_local_org_regain < 0 }
			}
			add_to_variable = { UKR_Insurgent_Army_local_org_regain = @subtraction_var }
			if = {
				limit = {
					check_variable = { UKR_Insurgent_Army_local_org_regain > 0 }
				}
				set_variable = { UKR_Insurgent_Army_local_org_regain = 0 }
			}
		}
		if = {
			limit = {
				check_variable = { UKR_Insurgent_Army_compliance_growth < 0 }
			}
			add_to_variable = { UKR_Insurgent_Army_compliance_growth = @subtraction_var }
			if = {
				limit = {
					check_variable = { UKR_Insurgent_Army_compliance_growth > 0 }
				}
				set_variable = { UKR_Insurgent_Army_compliance_growth = 0 }
			}
		}
		if = {
			limit = {
				check_variable = { UKR_Insurgent_Army_resistance_growth > 0 }
			}
			subtract_from_variable = { UKR_Insurgent_Army_resistance_growth = @subtraction_var }
			if = {
				limit = {
					check_variable = { UKR_Insurgent_Army_resistance_growth < 0 }
				}
				set_variable = { UKR_Insurgent_Army_resistance_growth = 0 }
			}
		}
		if = {
			limit = {
				check_variable = { UKR_Insurgent_Army_resistance_target > 0 }
			}
			subtract_from_variable = { UKR_Insurgent_Army_resistance_target = @subtraction_var }
			if = {
				limit = {
					check_variable = { UKR_Insurgent_Army_resistance_target < 0 }
				}
				set_variable = { UKR_Insurgent_Army_resistance_target = 0 }
			}
		}
		if = {
			limit = {
				check_variable = { UKR_Insurgent_Army_resistance_decay < 0 }
			}
			add_to_variable = { UKR_Insurgent_Army_resistance_decay = @subtraction_var }
			if = {
				limit = {
					check_variable = { UKR_Insurgent_Army_resistance_decay > 0 }
				}
				set_variable = { UKR_Insurgent_Army_resistance_decay = 0 }
			}
		}
		if = {
			limit = {
				check_variable = { UKR_Insurgent_Army_resistance_damage_to_garrison > 0 }
			}
			subtract_from_variable = { UKR_Insurgent_Army_resistance_damage_to_garrison = @subtraction_var }
			if = {
				limit = {
					check_variable = { UKR_Insurgent_Army_resistance_damage_to_garrison < 0 }
				}
				set_variable = { UKR_Insurgent_Army_resistance_damage_to_garrison = 0 }
			}
		}
	}
	else = {
		remove_dynamic_modifier = {
			modifier = UKR_Insurgent_Army
		}
		for_each_loop = {
			array = THIS.core_array
			THIS = {
				add_core_of = v
			}
			remove_from_array = {
				array = THIS.core_array
				value = v
			}
		}
		remove_core_of = UPA
	}
}

			# free_building_slots = { # more adaptive

			OR = {
				has_state_category = megalopolis
				has_state_category = metropolis
				has_state_category = large_city
				has_state_category = city
			}
			is_in_home_area = yes
			free_building_slots = {
				building = arms_factory
				size > 1
				include_locked = no
			}
			NOT = {
				has_dynamic_modifier = A_new_workforce
			}
		}
		add_extra_state_shared_building_slots = 1
		add_dynamic_modifier = {
			modifier = A_new_workforce
			days = 90
		}
	}
}

	divide_temp_variable = { temp_state_population_x2 = 100 }
