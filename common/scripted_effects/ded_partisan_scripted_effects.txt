@temp_x3 = 0.06
@temp_x1 = 0.02
set_UKR_Insurgent_Army_down = {
	if = {
		limit = {
			NOT = {
				check_variable = { UKR_Insurgent_Army_local_org_regain = 0 }
				check_variable = { UKR_Insurgent_Army_compliance_growth = 0 }
				check_variable = { UKR_Insurgent_Army_resistance_growth = 0 }
				check_variable = { UKR_Insurgent_Army_resistance_target = 0 }
				check_variable = { UKR_Insurgent_Army_resistance_decay = 0 }
				check_variable = { UKR_Insurgent_Army_resistance_damage_to_garrison = 0 }
			}
		}
		if = {
			limit = {
				check_variable = { UKR_Insurgent_Army_local_org_regain < 0 }
			}
			add_to_variable = { UKR_Insurgent_Army_local_org_regain = @temp_x1 }
		}
		if = {
			limit = {
				check_variable = { UKR_Insurgent_Army_compliance_growth < 0 }
			}
			add_to_variable = { UKR_Insurgent_Army_compliance_growth = @temp_x1 }
		}
		if = {
			limit = {
				check_variable = { UKR_Insurgent_Army_resistance_growth > 0 }
			}
			subtract_from_variable = { UKR_Insurgent_Army_resistance_growth = @temp_x1 }
		}
		if = {
			limit = {
				check_variable = { UKR_Insurgent_Army_resistance_target > 0 }
			}
			subtract_from_variable = { UKR_Insurgent_Army_resistance_target = @temp_x3 }
		}
		if = {
			limit = {
				check_variable = { UKR_Insurgent_Army_resistance_decay < 0 }
			}
			add_to_variable = { UKR_Insurgent_Army_resistance_decay = @temp_x3 }
		}
		if = {
			limit = {
				check_variable = { UKR_Insurgent_Army_resistance_damage_to_garrison > 0 }
			}
			subtract_from_variable = { UKR_Insurgent_Army_resistance_damage_to_garrison = @temp_x1 }
		}
	}
	else = {
		remove_dynamic_modifier = {
			modifier = UKR_Insurgent_Army
		}
		add_core_of = UKR
		if = {
			limit = {
				is_UKR_Insurgent_army_POL = yes
			}
			add_core_of = POL
		}
		remove_core_of = UPA
	}
}
